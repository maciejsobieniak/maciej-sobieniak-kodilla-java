SELECT IL.NAME AS ISSUES_LIST_NAME, I.TITLE, I.DESCRIPTION FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.TITLE, I.DESCRIPTION FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNED = U.ID;

SELECT U.ID AS USER_ID, I.TITLE, I.DESCRIPTION FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNED = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUSE_NUMBER
FROM USERS U JOIN ISSUES I ON I.USER_ID_ASSIGNED = U.ID
GROUP BY U.ID
HAVING ISSUSE_NUMBER >= 1;
